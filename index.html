const OPTION_LABELS = ['a', 'b', 'c', 'd'];

let current = 0;

let userAnswers = new Array(50).fill(-1);

let marked = new Array(50).fill(false);

let visited = new Array(50).fill(false);

let timerInterval;

let timeLeft = 60 * 60;

let submitted = false;


// Build Q Grid

function buildGrid() {

  const grid = document.getElementById('q-grid');

  grid.innerHTML = '';

  for (let i = 0; i < 50; i++) {

    const btn = document.createElement('button');

    btn.className = 'q-btn';

    btn.textContent = i + 1;

    btn.id = `qb-${i}`;

    btn.onclick = () => goTo(i);

    grid.appendChild(btn);

  }

}


function updateGridBtn(i) {

  const btn = document.getElementById(`qb-${i}`);

  btn.className = 'q-btn';

  if (i === current) btn.classList.add('current');

  else if (marked[i] && userAnswers[i] !== -1) btn.classList.add('answered', 'marked');

  else if (marked[i]) btn.classList.add('marked');

  else if (userAnswers[i] !== -1) btn.classList.add('answered');

}


function updateAllGridBtns() {

  for (let i = 0; i < 50; i++) updateGridBtn(i);

}


function updateStats() {

  const answered = userAnswers.filter(a => a !== -1).length;

  const markedCount = marked.filter(m => m).length;

  document.getElementById('stat-ans').textContent = answered;

  document.getElementById('stat-not').textContent = 50 - answered;

  document.getElementById('stat-mark').textContent = markedCount;

}


function renderQuestion(i) {

  visited[i] = true;

  const q = QUESTIONS[i];

  document.getElementById('q-label').textContent = `Q.${i + 1}`;

  document.getElementById('q-text').innerHTML = q.q.replace(/\n/g, '<br>');


  const optsList = document.getElementById('options-list');

  optsList.innerHTML = '';

  q.opts.forEach((opt, idx) => {

    const li = document.createElement('li');

    li.className = 'option-item' + (userAnswers[i] === idx ? ' selected' : '');

    li.innerHTML = `<span class="opt-key">${OPTION_LABELS[idx]})</span><span class="opt-text">${opt}</span>`;

    li.onclick = () => selectOption(idx);

    optsList.appendChild(li);

  });


  // Mark button

  const markBtn = document.getElementById('btn-mark');

  markBtn.classList.toggle('active', marked[i]);

  markBtn.textContent = marked[i] ? '🔖 Marked' : '🔖 Mark for Review';


  updateAllGridBtns();

  updateStats();

  document.getElementById('question-panel').scrollTop = 0;

}


function selectOption(idx) {

  if (submitted) return;

  userAnswers[current] = idx;

  renderQuestion(current);

}


function saveAndNext() {

  if (current < 49) { current++; renderQuestion(current); }

}


function navigate(dir) {

  const next = current + dir;

  if (next >= 0 && next < 50) { current = next; renderQuestion(current); }

}


function goTo(i) {

  current = i;

  renderQuestion(i);

}


function toggleMark() {

  marked[current] = !marked[current];

  renderQuestion(current);

}


function clearAnswer() {

  userAnswers[current] = -1;

  renderQuestion(current);

}


// TIMER

function startTimer() {

  timerInterval = setInterval(() => {

    timeLeft--;

    const m = Math.floor(timeLeft / 60);

    const s = timeLeft % 60;

    const el = document.getElementById('timer');

    el.textContent = `${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;

    if (timeLeft <= 300) el.classList.add('danger');

    if (timeLeft <= 0) { clearInterval(timerInterval); submitTest(); }

  }, 1000);

}


function openConfirm() {

  const answered = userAnswers.filter(a => a !== -1).length;

  const unanswered = 50 - answered;

  document.getElementById('confirm-msg').textContent =

    `You have answered ${answered} out of 50 questions. ${unanswered > 0 ? unanswered + ' question(s) are unanswered.' : ''} Are you sure you want to submit?`;

  document.getElementById('confirm-modal').classList.add('active');

}

function closeConfirm() { document.getElementById('confirm-modal').classList.remove('active'); }


function submitTest() {

  closeConfirm();

  clearInterval(timerInterval);

  submitted = true;

  showResult();

}


function showResult() {

  let correct = 0, wrong = 0, skip = 0;

  QUESTIONS.forEach((q, i) => {

    if (userAnswers[i] === -1) skip++;

    else if (userAnswers[i] === q.ans) correct++;

    else wrong++;

  });

  document.getElementById('res-score').textContent = correct;

  document.getElementById('res-correct').textContent = correct;

  document.getElementById('res-wrong').textContent = wrong;

  document.getElementById('res-skip').textContent = skip;


  const reviewList = document.getElementById('review-list');

  reviewList.innerHTML = '';

  QUESTIONS.forEach((q, i) => {

    const userA = userAnswers[i];

    const isCorrect = userA === q.ans;

    const isSkip = userA === -1;

    const div = document.createElement('div');

    div.className = 'review-item';

    const qShort = q.q.split('\n')[0].substring(0, 80) + (q.q.split('\n')[0].length > 80 ? '...' : '');

    let ansText, ansClass;

    if (isSkip) { ansText = 'Skipped'; ansClass = 'skip'; }

    else if (isCorrect) { ansText = `✓ (${OPTION_LABELS[userA]})`; ansClass = 'correct'; }

    else { ansText = `✗ ${OPTION_LABELS[userA]} → ${OPTION_LABELS[q.ans]}`; ansClass = 'wrong'; }

    div.innerHTML = `<span class="rv-num">Q${i+1}</span><span class="rv-q">${qShort}</span><span class="rv-ans ${ansClass}">${ansText}</span>`;

    reviewList.appendChild(div);

  });


  document.getElementById('result-screen').style.display = 'block';

}


function restartTest() {

  userAnswers = new Array(50).fill(-1);

  marked = new Array(50).fill(false);

  visited = new Array(50).fill(false);

  current = 0;

  submitted = false;

  timeLeft = 60 * 60;

  document.getElementById('timer').textContent = '60:00';

  document.getElementById('timer').classList.remove('danger');

  document.getElementById('result-screen').style.display = 'none';

  buildGrid();

  renderQuestion(0);

  startTimer();

}

renderQuestion(0);

startTimer();
